## 一、前后端交互原理

### 1.3 HTTP 请求方法

| 方法 | 作用 | 示例 |
| --- | --- | --- |
| **GET** | 查询数据 | 获取图书列表、获取读者信息 |
| **POST** | 创建数据 | 登录、注册、借书、上架图书 |
| **PUT** | 更新数据 | 修改手机号、还书、重置密码 |
| **DELETE** | 删除数据 | 下架图书 |

### 1.4 统一响应格式

后端所有接口返回相同格式：

```json
{
  "code": 0,
  "message": "success",
  "data": {
    // 业务数据
  }
}
```

| code | 含义 |
| --- | --- |
| **0** | 成功 |
| **4001** | 存在超期记录，禁止借书 |
| **4002** | 已达到最大借阅数量上限 |
| **4003** | 库存不足 |
| **4004** | 读者不存在 |
| **4005** | 图书不存在 |
| **4006** | 读者已冻结 |
| **4007** | 参数校验失败 |
| **4008** | 证号已被注册 |
| **4009** | 重复借阅 |
| **403** | 无权限访问 |
| **500** | 服务器内部错误 |

### 2.2 前端代码

```vue
<!-- LoginView.vue -->
<template>
  <div class="login-container">
    <div class="login-form">
      <h2>图书馆管理系统</h2>
      
      <EvoInput
        v-model="form.readerId"
        label="借阅证号"
        placeholder="请输入借阅证号"

      />

        @click="handleLogin"

      </EvoButton>

<script setup>
import { ref, reactive } from 'vue'
import { useAuthStore } from '@/stores/auth'
import { useRouter } from 'vue-router'

import EvoButton from '@/components/EvoButton.vue'

const authStore = useAuthStore()
const router = useRouter()
const loading = ref(false)
const errorMsg = ref('')

  readerId: '',
  password: ''
})

async function handleLogin() {
  errorMsg.value = ''
  
  if (!form.readerId.trim()) {
    errorMsg.value = '请输入借阅证号'
    return

    return

    await authStore.login(form.readerId.trim(), form.password.trim())
    router.push(authStore.homePath)
  } catch (error) {
    errorMsg.value = error.message || '登录失败'
  } finally {
    loading.value = false
  }
}
</script>
```

### 2.3 登录流程详解

#### 阶段 1：用户操作

```
用户在登录页面：
1. 输入借阅证号："R001"
2. 输入密码："123456"
3. 点击「登录」按钮

触发事件：@click="handleLogin"
```

#### 阶段 2：前端处理

```javascript
// handleLogin() 方法执行
async function handleLogin() {
  // Step 1: 前端校验
  if (!form.readerId.trim()) {
    errorMsg.value = '请输入借阅证号'
    return
  }
  if (!form.password.trim()) {
    errorMsg.value = '请输入密码'
    return
  }
  
  // Step 2: 设置 loading 状态
  loading.value = true
  
  // Step 3: 调用 Pinia Store 的 login 方法
  await authStore.login(form.readerId.trim(), form.password.trim())
}
```

#### 阶段 3：Pinia Store

```javascript
// stores/auth.js
async function login(readerId, password) {
  // Step 4: Axios 发送 POST 请求
  const response = await api.post('/auth/login', { readerId, password })
  
  // Step 5: 保存 Token 和用户信息
  token.value = response.data.token
  role.value = response.data.role
  readerName.value = response.data.readerName
  
  // Step 6: 持久化到 localStorage
  localStorage.setItem('token', token.value)
  localStorage.setItem('role', role.value)
}
```

#### 阶段 4：Axios 请求拦截

```javascript
// 请求拦截器自动添加 Token
api.interceptors.request.use(config => {
  const authStore = useAuthStore()
  if (authStore.token) {
    config.headers.Authorization = `Bearer ${authStore.token}`
  }
  return config
})

// 此时 Token 为空（首次登录），所以不添加 Authorization 头
```

#### 阶段 5：Vite 代理转发

```
前端请求：http://localhost:3000/api/v1/auth/login
Vite 代理：/api → http://localhost:8080
后端接收：http://localhost:8080/api/v1/auth/login
```

#### 阶段 6：后端 SecurityConfig

```java
// SecurityConfig.java
.authorizeHttpRequests()
    .antMatchers("/api/v1/auth/login").permitAll()  // 登录接口公开访问
    .anyRequest().authenticated()
```

登录接口配置为 `permitAll()`，所以直接放行，不需要 Token 校验。

#### 阶段 7：Controller 层

```java
// AuthController.java
@PostMapping("/login")
public Result<LoginResponse> login(@Valid @RequestBody LoginRequest request) {
    LoginResponse response = authService.login(request);
    return Result.ok(response);
}
```

- `@Valid` 触发参数校验（readerId、password 不能为空）
- 调用 `AuthService.login()` 处理业务逻辑

#### 阶段 8：Service 层

```java
// AuthService.java
public LoginResponse login(LoginRequest request) {
    // Step 8.1: 查询读者
    Reader reader = readerMapper.selectById(request.getReaderId());
    
    // Step 8.2: 校验读者存在性
    if (reader == null) {
        throw new BusinessException(ErrorCode.READER_NOT_FOUND);
    }
    
    // Step 8.3: BCrypt 密码比对
    if (!passwordEncoder.matches(request.getPassword(), reader.getPassword())) {
        throw new BusinessException(ErrorCode.LOGIN_FAILED);
    }
    
    // Step 8.4: 生成 JWT Token
    String token = jwtTokenProvider.generateToken(reader.getId(), reader.getRole());
    
    // Step 8.5: 返回响应
    return LoginResponse.builder()
        .token(token)
        .role(reader.getRole())
        .readerName(reader.getName())
        .readerId(reader.getId())
        .build();
}
```

```java
// ReaderMapper.java
```

```vue
const loading = ref(false)

const columns = [

async function handleBorrow() {
  message.value = ''
  
  if (!form.readerId.trim()) {
    message.value = '请输入读者 ID'

    message.value = '请输入 ISBN'
    messageType.value = 'error'
    return

    await api.post('/borrow-records', {
      readerId: form.readerId.trim(),

    messageType.value = 'error'

    loading.value = false

async function loadBorrowList() {
  if (!form.readerId.trim()) return

  borrowList.value = response.data
```

```java
// BorrowService.java
@Transactional(rollbackFor = Exception.class)
public BorrowRecordDTO borrow(BorrowRequest request) {
    // Step 9.1: 校验读者存在性
    Reader reader = readerMapper.selectById(request.getReaderId());
    if (reader == null) {
        throw new BusinessException(ErrorCode.READER_NOT_FOUND);  // 4004
    }
    
    // Step 9.2: 检查超期状态
    long overdueCount = borrowRecordMapper.countOverdueByReaderId(request.getReaderId());
    if (overdueCount > 0) {
        throw new BusinessException(ErrorCode.HAS_OVERDUE);  // 4001
    }
    
    // Step 9.3: 检查借阅上限
    if (reader.getCurrentBorrowCount() >= reader.getMaxBorrowCount()) {
        throw new BusinessException(ErrorCode.EXCEED_MAX_BORROW);  // 4002
    }
    
    // Step 9.4: 校验图书存在性
    Book book = bookMapper.selectById(request.getIsbn());
    if (book == null || !book.getIsActive()) {
        throw new BusinessException(ErrorCode.BOOK_NOT_FOUND);  // 4005
    }
    
    // Step 9.5: 扣库存（行锁防超卖）
    int updateCount = bookMapper.decreaseStock(request.getIsbn());
    if (updateCount == 0) {
        throw new BusinessException(ErrorCode.STOCK_INSUFFICIENT);  // 4003
    }
    
    // Step 9.6: 检查重复借阅
    BorrowRecord existing = borrowRecordMapper.selectByReaderAndBook(
        request.getReaderId(), request.getIsbn());
    if (existing != null && "BORROWED".equals(existing.getStatus())) {
        throw new BusinessException(ErrorCode.DUPLICATE_BORROW);  // 4009
    }
    
    // Step 9.7: 创建借阅记录
    BorrowRecord record = new BorrowRecord();
    record.setReaderId(request.getReaderId());
    record.setIsbn(request.getIsbn());
    record.setBorrowDate(LocalDate.now());
    record.setDueDate(LocalDate.now().plusDays(30));
    record.setStatus("BORROWED");
    borrowRecordMapper.insert(record);
    
    // Step 9.8: 更新读者借阅计数
    readerMapper.increaseBorrowCount(request.getReaderId());
    
    // Step 9.9: 写入审计日志
    auditLogMapper.insert(AuditLog.builder()
        .operatorId(request.getReaderId())
        .action("BORROW")
        .targetType("BOOK")
        .targetId(request.getIsbn())
        .detail("读者[R001]借阅图书[9787111544937]")
        .build());
    
    return convertToDTO(record, book, reader);
}
```

## 四、前后端交互核心要点总结

| 技术点 | 是什么 | 为什么用 | 凭什么这么做 |
| --- | --- | --- | --- |
| RESTful API | HTTP 协议 + 统一 URL 规范 | 前后端解耦，易于测试和文档化 | HTTP 标准方法 + JSON 格式 |
| JWT Token | 无状态认证方案 | 不需要服务器保存会话，支持水平扩展 | Token 自包含用户信息 |
| Axios 拦截器 | 请求/响应统一处理 | 自动添加 Token，统一错误处理 | Promise 链式调用 |
| Vite 代理 | 开发环境跨域解决方案 | 浏览器同源策略限制，代理转发请求 | Node.js http-proxy |
| @Transactional | Spring 声明式事务 | 多条 SQL 原子操作，保证数据一致性 | Spring AOP 拦截方法 |
| @Valid | 参数校验 | 自动校验请求参数，减少手动判断 | Jakarta Validation 标准 |
| Result\<T\> | 统一响应格式 | 前端处理统一，便于错误处理 | 所有接口返回相同结构 |
| Pinia Store | 全局状态管理 | 多个组件共享状态，登录信息持久化 | Composition API + localStorage |
